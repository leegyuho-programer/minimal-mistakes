```javascript
arr.sort()
// 배열 재정렬
// 배열 자체가 변경되니 주의
// 인수로 정렬 로직을 담은 함수를 받음

let arr = [1,5,4,2,3];

arr.sort();

console.log(arr); // [1,2,3,4,5]



let arr = ["a","c","d","e","b"]

arr.sort();

console.log(arr); // ["a","b","c","d","e"]



let arr = [27,8,5,13];

arr.sort();

console.log(arr); //[13,27,5,8]
//앞 숫자가 1이고 2라서 13과 27이 앞에 왔다. 정렬할때 문자열로 취급하기 때문이다.



let arr = [27,8,5,13];

function fn(a,b){
  return a - b;
}//두개씩 뽑아서 비교하여 작은 것을 앞으로 가게하는 것이다.

arr.sort(fn);

console.log(arr); // [5,8,13,27] 이렇게 하면 된다.

arr.sort((a,b) => {
  return a - b;
}); //이렇게 해도 된다.

//Lodash (원하는 기준으로 정렬시켜준다.)

arr.reduce()
//인수로 함수를 받음

//초기값은 안써도 된다. 안쓰게 되면 배열의 첫번째 값이 들어간다.

  return prev + cur;

console.log(result);



let userList = [

  { name: "Tom", age: 10 },
```

구조 분해 할당 (Destructuring assignment)
```javascript
Desturcturing assignment
//구조 분해 할당 구문은 배열이나 객체의 속성을 분해해서
//그 값을 변수에 담을 수 있게 하는 표현

배열 구조 분해

let [x, y] = [1, 2];

console.log(x); // 1
console.log(y); // 2


let users = ['Mike', 'Tom', 'Jane'];
let [user1, user2, user3] = users;
//let user1 = users[0];
//let user2 = users[1];
//let user3 = users[2];

console.log(user1); // 'Mike'
console.log(user2); // 'Tom'
console.log(user3); // 'Jane'



let str = "Mike-Tom-Jane";
let [user1, user2, user3] = str.split('-');
//str.split('-') : ["Mike", "Tom", "Jane"]

console.log(user1); // 'Mike'
console.log(user2); // 'Tom'
console.log(user3); // 'Jane'






배열 구조 분해 : 기본값

let [a,b,c] = [1,2];
//c = undefined



let [a=3, b=4, c=5] = [1,2];

console.log(a); // 1
console.log(b); // 2
console.log(c); // 5






배열 구조 분해 : 일부 반환값 무시

let [user1, , user2] = ['Mike', 'Tom', 'Jane', 'Tony'];

console.log(user1); //'Mike'
console.log(user2); //'Jane'






배열 구조 분해 : 바꿔치기

let a = 1;
let b = 2;

//a = b; XXXXXXX 이렇게 하면 a = 2, b = 2 이렇게 된다

let c = a;
a = b;
b = c;
// a = 2, b = 1

[a, b] = [b, a]; // 이렇게 간단하게 할 수도 있다.






객체 구조 분해

let user = {name: 'Mike', age: 30};
let {name, age} = user;

console.log(name); //'Mike'
console.log(age); //30



let user = {name: 'Mike', age: 30};
let {name: userName, age: userAge} = user;

console.log(userName); //'Mike'
console.log(userAge); //30






객체 구조 분해 : 기본값

let user = {name: 'Mike', age: 30};
let {name, age, gender} = user; // gender : undefined
let {name, age, gender = 'male'} = user; //gender를 기본값으로 할당하는 것

let user = {
  name: 'Jane',
  age: 18,
  gender: 'female',
};

let {name, age, gender = 'male'} = user;

console.log(gender); //'female'
//user의 gender에 아무 값도 할당되지 않았다면 male이라고 나왔다
```
