css만들기 연습
```c++
/*연습 1*/
<!DOCTYPE html>
<html>
  <head>
     <meta charset="utf-8">
     <meta name="author" content="egoing">
      <style>
        .box {
          border:50px solid red;
          padding:50px;
          width:200px;
          margin-bottom:20px;
        }
        
        .green {
          border-color:green;
          /* display: none; 아이에 사라지고 그 자리에 밑에 있는 것이 올라옴*/
          /* visibility: hidden; 자리가 비어있음*/
        }
        
        .blue {
          border-color:blue;
        }
      </style>
  </head>
  <body>
    <div class="box">Box 1</div>
    <div class="box green">Box 2</div>
    <div class="box blue">Box 3</div>
  </body>
  </html>

/*연습 2*/
```
